#include<stdio.h>
void bubble_sort (int arr[] , int n){
for( int i=0; i<n ; i++){
    for(int j=0 ; j<n-i-1; j++){
      if (arr[j] > arr[j+1]){
      int temp = arr[j];
      arr[j] = arr[j+1];
      arr[j+1] = temp;
    }
    }
}
}

    int arr []={2,10,3,90,1};
    bubble_sort(arr , 5);
